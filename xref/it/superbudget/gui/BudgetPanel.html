449 	private void calculate()
450 	{
451 		dataset.clear();
452 		lblBudgetName.setText(budget.getName());
453 		NumberFormat format = NumberFormat.getCurrencyInstance();
454 		labelEntrateAnnuali.setText(this.calculateEntrateAnnuali());
455 		labelSpeseAnnuali.setText(this.calculateSpeseAnnuali());
456 		BigDecimal saldo = this.calculateSaldoParziale();
457 		format.setMaximumFractionDigits(3);
458 		lblSaldoParziale.setText(format.format(saldo));
459 		if (saldo.signum() == -1)
460 		{
461 			lblSaldoParziale.setForeground(Color.RED);
462 		}
463 		else
464 		{
465 			lblSaldoParziale.setForeground(Color.green);
466 		}
467 		saldo = budget.getSaldo().add(saldo);
468 		labelSaldo.setText(format.format(saldo));
469 		if (budget.getSaldo().signum() == -1)
470 		{
471 			labelSaldo.setForeground(Color.red);
472 		}
473 		else
474 		{
475 			labelSaldo.setForeground(Color.green);
476 		}
477 		try
478 		{
479 			dataset.setValue("Income", format.parse(labelEntrateAnnuali.getText()));
480 			dataset.setValue("Expense", format.parse(labelSpeseAnnuali.getText()));
481 		}
482 		catch (ParseException e)
483 		{
484 			// TODO Auto-generated catch block
485 			e.printStackTrace();
486 		}
487 	}
488 
489 	private BigDecimal calculateSaldoParziale()
490 	{
491 		try
492 		{
493 			NumberFormat format = NumberFormat.getCurrencyInstance();
494 			Number number = format.parse(this.labelEntrateAnnuali.getText());
495 			BigDecimal entrate = new BigDecimal(number.doubleValue());
496 			number = format.parse(this.labelSpeseAnnuali.getText());
497 			BigDecimal spese = new BigDecimal(number.doubleValue());
498 
499 			BigDecimal saldo = entrate.subtract(spese);
500 			return saldo;
501 		}
502 		catch (ParseException e)
503 		{
504 			logger.error("Calculate Saldo Parziale generate an error", e);
505 			MessagesUtils.showErrorMessage(this, labels.getString(BUDGET_PANEL_LABEL_CALCULATE_ERROR));
506 			return new BigDecimal(0.0);
507 		}
508 	}
